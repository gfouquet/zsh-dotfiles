Oh my zsh

```
sh -c "$(curl -fsSL https://raw.github.com/ohmyzsh/ohmyzsh/master/tools/install.sh)"
```

Source dotfiles

```
echo >>~/.zshrc <<EOF
for file in ~/.dotfiles/*(.); do
  if [[ ! ${file} =~ \.md ]]; then
      source ${file}
  fi
done
EOF

```
